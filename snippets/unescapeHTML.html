Unescapes escaped HTML characters.

  • Use String.prototype.replace() with a regexp that matches the characters that need to be unescaped.
  • Use the functionâ€™s callback to replace each escaped character instance with its associated unescaped character using a dictionary (object).
const unescapeHTML = str =>
  str.replace(
    /&amp;|&lt;|&gt;|&#39;|&quot;/g,
    tag =>
      ({
        '&amp;': '&',
        '&lt;': '<',
        '&gt;': '>',
        '&#39;': "'",
        '&quot;': '"'
      }[tag] || tag)
  );
unescapeHTML('&lt;a href=&quot;#&quot;&gt;Me &amp; you&lt;/a&gt;');
// '<a href="#">Me & you</a>'
